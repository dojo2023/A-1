create table user_information(
USER_ID INT AUTO_INCREMENT NOT NULL ,
USER_NAME VARCHAR(8) NOT NULL,
USER_SEX INT NOT NULL,
USER_BIRTH DATE NOT NULL,
USER_MAIL_ADDRESS VARCHAR(50) NOT NULL,
USER_PASSWORD VARCHAR(20) NOT NULL,
USER_HEIGHT INT NOT NULL,
USER_WEIGHT INT NOT NULL,
PRIMARY KEY (USER_ID)
)

CREATE TABLE TRAINING_MENU(
TRAINING_MENU_ID INT NOT NULL,
TRAINING_MENU VARCHAR NOT NULL,
TRAINING_MENU_MAGNIFICATION DOUBLE NOT NULL,
PRIMARY KEY (TRAINING_MENU_ID)
)

CREATE TABLE TRAINING_RECORD(
TRAINING_RECORD_ID INT AUTO_INCREMENT NOT NULL,
TRAINING_RECORD_DATE DATE NOT NULL,
USER_ID INT NOT NULL,
TRAINING_MENU VARCHAR NOT NULL,
TRAINING_WEIGHT DOUBLE NOT NULL,
TRAINING_COUNT INT NOT NULL,
TRAINING_SET INT NOT NULL,
TRAINING_EXP INT NOT NULL,

PRIMARY KEY(TRAINING_RECORD_ID),
FOREIGN KEY(USER_ID)
	REFERENCES USER_INFORMATION(USER_ID),
FOREIGN KEY(TRAINING_MENU)
	REFERENCES TRAINING_MENU(TRAINING_MENU)
)
